    public override void Load()
    {
        // IL_5197: call         void Terraria.Main::ClearPendingPlayerSelectCallbacks()
        // IL_519c: ldc.i4.1
        // IL_519d: stsfld       int32 Terraria.Main::menuMode
        IL.Terraria.Main.DrawMenu += Skip;
    }

    private void Skip(ILContext il)
    {
        var c = new ILCursor(il);
        c.GotoNext(i => i.MatchCall(typeof(Main), nameof(Main.ClearPendingPlayerSelectCallbacks)));
        // Join via IP
        c.GotoNext(i => i.MatchCall(typeof(Main), nameof(Main.ClearPendingPlayerSelectCallbacks)));
        c.GotoNext(i => i.MatchStsfld(typeof(Main), nameof(Main.menuMode)));
        c.EmitDelegate<Func<byte, byte>>(_ => { return 13; });
        // Host & Play
        c.GotoNext(i => i.MatchCall(typeof(Main), nameof(Main.ClearPendingPlayerSelectCallbacks)));
        c.GotoNext(i => i.MatchStsfld(typeof(Main), nameof(Main.menuMode)));
        c.EmitDelegate<Func<byte, byte>>(_ =>
        {
            Main.SelectPlayer(new PlayerFileData()
            {
                Player = new Player()
                {
                    name = "Guest",
                }
            });
            return 6;
        });
    }

    public override void Unload()
    {
        IL.Terraria.Main.DrawMenu -= Skip;
    }
